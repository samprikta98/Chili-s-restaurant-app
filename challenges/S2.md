## Context

Angular Signals (stable since Angular 17) offer a simpler, synchronous alternative to RxJS for local and shared UI state. This challenge replaces the `BehaviorSubject`-based order status with signals, and introduces `computed` and `effect` in the menu cart.

## Current code

**`order-tracking.service.ts`:**
```typescript
export class OrderTrackingService {
  private orderStatus = new BehaviorSubject<string>('Not started');
  orderStatus$ = this.orderStatus.asObservable();
  private trackingSubscription: Subscription | undefined;

  statuses = ['Preparing', 'Cooking', 'Quality Check', 'Out for Delivery', 'Delivered'];

  startTracking() {
    this.stopTracking();
    this.trackingSubscription = interval(5000).subscribe(() => {
      const randomStatus = this.statuses[Math.floor(Math.random() * this.statuses.length)];
      this.orderStatus.next(randomStatus);
    });
  }

  stopTracking() {
    if (this.trackingSubscription) {
      this.trackingSubscription.unsubscribe();
    }
  }
}
```

**`order-tracker.component.ts`:**
```typescript
export class OrderTrackerComponent implements OnInit {
  currentStatus: string = 'Not started';

  ngOnInit() {
    this.statusSubscription = this.orderTrackingService.orderStatus$.subscribe(
```
